public class Student
{
	public int StudentID {get; set;}
	public string StudentName {get; set;}

	public Student(int studentId, string studentName)
	{
		StudentID = studentId;
		StudentName = studentName;
	}

	public override int Equals(Student student)
	{
		Student other = student as Student;
                
		if (other == null)
                return false;

		return this.StudentID == other.StudentID && this.StudentName = this.StudentName;
	}

	public override int GetHashCode()
	{
		return studentID.GetHashCode();
	}
}

public class StudentMarks
{

	Dictionary<Student, int> StudentRecords = new Dictionary<Student, int>();
	
	pubic void AddStudentRecord(Student student, int marks)
	{
		if(StudentRecords.ContainsKey(student))
		{
			if(marks > StudentRecords[student])
			{
				StudentRecords[student] = marks;
			}
		}
		else
		{
			StudentRecords.Add(student,marks);
		}
	}




}
